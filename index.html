async function fetchStockPrice() { let input = document.getElementById('scripInput').value.trim().toUpperCase(); const resultDiv = document.getElementById('result'); const loadingStatus = document.getElementById('loadingStatus'); const stockTitle = document.getElementById('stockTitle'); const stockPrice = document.getElementById('stockPrice'); const stockDetails = document.getElementById('stockDetails'); if (!input) return; // Convert standard inputs to the format GoogleFinance expects for BSE (e.g., BOM:500325 or BOM:RELIANCE) let bseTicker = input.startsWith("BOM:") ? input : `BOM:${input}`; resultDiv.style.display = 'none'; loadingStatus.style.display = 'block'; // PASTE YOUR GOOGLE APPS SCRIPT WEB APP URL HERE const googleAppUrl = `https://script.google.com/macros/s/AKfycbwWLHuLdM6H8DD3Obci0w171fArUVMHxUUUIxcui502tJQsWRBw2Cg0R-khviYXxT9A8g/exec?ticker=${encodeURIComponent(bseTicker)}`; try { const response = await fetch(googleAppUrl); const data = await response.json(); if (!data.success) throw new Error('Invalid Stock Symbol'); loadingStatus.style.display = 'none'; resultDiv.style.display = 'block'; resultDiv.className = 'result-box success'; stockTitle.innerText = input; // Formatted cleanly using the Indian Numbering System stockPrice.innerText = `₹ ${data.price.toLocaleString('en-IN', { minimumFractionDigits: 2 })}`; const direction = data.change >= 0 ? '▲' : '▼'; stockDetails.innerText = `${direction} ${Math.abs(data.change).toFixed(2)} (${data.pct.toFixed(2)}%) Today`; stockDetails.style.color = data.change >= 0 ? '#166534' : '#991b1b'; } catch (err) { loadingStatus.style.display = 'none'; resultDiv.style.display = 'block'; resultDiv.className = 'result-box error'; stockTitle.innerText = 'Error'; stockPrice.innerText = ''; stockDetails.innerText = 'Unable to fetch data. Ensure you typed a valid BSE Scrip name or 6-digit text code.'; stockDetails.style.color = '#991b1b'; } }
